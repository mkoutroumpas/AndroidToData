Use Java Web runtime library, Dynamic Web Module v.2.5, Java 1.7 facet and JavaEE 6 Web Profile/JRE7 on https://account.hanatrial.ondemand.com/cockpit 
Deployment descriptor must be:

<?xml version="1.0" encoding="UTF-8"?>
<web-app xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns="http://java.sun.com/xml/ns/javaee"
    xmlns:web="http://java.sun.com/xml/ns/javaee/web-app_2_5.xsd" xsi:schemaLocation="http://java.sun.com/xml/ns/javaee http://java.sun.com/xml/ns/javaee/web-app_2_5.xsd"
    id="WebApp_ID" version="2.5">

  <servlet>
    <servlet-name>DataAccessServlet</servlet-name>
    <servlet-class>com.mk.edu.DataAccessServlet</servlet-class>
  </servlet>
  <servlet-mapping>
    <servlet-name>DataAccessServlet</servlet-name>
    <url-pattern>/</url-pattern>
  </servlet-mapping>
</web-app>
